32.1.3 Trending: Extreme value

toplot2 <- aqi_data %>%
    arrange(日期)%>%
    filter(測項=="PM2.5") %>%
    gather("hour", "PM25", 4:28) %>%
    mutate(PM25 = as.numeric(PM25)) %>%
    drop_na() %>%
    group_by(日期) %>%
    summarize(avg = sum(PM25)/24) %>% 
    ungroup() %>%
    mutate(year = lubridate::year(日期), month = lubridate::month(日期)) %>%
    group_by(year, month) %>%
    summarize(purple = sum(avg>150),
              red = sum(avg>54),
              orange = sum(avg>35)) %>%
    ungroup()
